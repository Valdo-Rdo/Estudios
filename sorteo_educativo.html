import React, { useState } from 'react';

import { Users, GraduationCap, Sparkles, Loader2 } from 'lucide-react';


const ESTUDIANTES = [

  "PAULA SOFIA ARCEO GOMEZ",

  "JUANITA GUADALUPE CASTRO LOPEZ",

  "ANDREA GOMEZ DELGADILLO",

  "TATIANA VANESSA LOPEZ GARCIA",

  "GABRIELA ESTEFANIA LOPEZ VILLANUEVA",

  "LIDIA ESTEFANIA MEZA SEPULVEDA",

  "XIMENA MARGARITA NAVARRO MELGAREJO",

  "TERESA DE JESUS RODRIGUEZ BERMUDES",

  "ANGELICA RODRIGUEZ MAGDALENO",

  "LAURA RODRIGUEZ MONTERO",

  "OSWALDO RODRIGUEZ MONTERO"

];


const NIVELES = [

  "Multigrado",

  "Posgrado",

  "Secundaria",

  "Primaria Mayor",

  "Preescolar",

  "Primaria menor",

  "Preparatoria",

  "Capacitación",

  "Licenciatura",

  "Educación continua libre (Plataformas)",

  "Virtual Asincrónico"

];


// Función para mezclar arreglos (Fisher-Yates Shuffle)

const shuffleArray = (array) => {

  const newArray = [...array];

  for (let i = newArray.length - 1; i > 0; i--) {

    const j = Math.floor(Math.random() * (i + 1));

    [newArray[i], newArray[j]] = [newArray[j], newArray[i]];

  }

  return newArray;

};


export default function App() {

  const [asignaciones, setAsignaciones] = useState([]);

  const [isGenerating, setIsGenerating] = useState(false);


  const generarAsignaciones = () => {

    setIsGenerating(true);

    setAsignaciones([]); // Limpiar anteriores


    // Simulamos un pequeño tiempo de carga para el efecto visual

    setTimeout(() => {

      // Mezclamos aleatoriamente los niveles

      const nivelesMezclados = shuffleArray(NIVELES);


      // Creamos la lista final uniendo al estudiante con un nivel

      const nuevasAsignaciones = ESTUDIANTES.map((estudiante, index) => ({

        id: index,

        nombre: estudiante,

        nivel: nivelesMezclados[index] // Como hay 11 niveles y 11 alumnos, nadie repite nivel

      }));


      setAsignaciones(nuevasAsignaciones);

      setIsGenerating(false);

    }, 800);

  };


  return (

    <div className="min-h-screen bg-slate-100 p-4 md:p-8 font-sans text-slate-800">

      <div className="max-w-6xl mx-auto space-y-8">

        

        {/* Encabezado */}

        <div className="bg-white rounded-3xl p-8 shadow-sm border border-slate-200 text-center relative overflow-hidden">

          <div className="absolute top-0 left-0 w-full h-2 bg-gradient-to-r from-blue-500 via-indigo-500 to-purple-500"></div>

          <h1 className="text-3xl md:text-4xl font-extrabold text-slate-800 mb-4">

            Asignación de Proyectos a Grupo

          </h1>

          <p className="text-slate-500 max-w-2xl mx-auto mb-8 text-lg">

            Asigna automáticamente un nivel educativo único al azar para cada uno de los {ESTUDIANTES.length} estudiantes del grupo.

          </p>

          

          <button

            onClick={generarAsignaciones}

            disabled={isGenerating}

            className={`inline-flex items-center space-x-2 px-8 py-4 rounded-full text-white font-bold text-lg transition-all duration-200 shadow-lg hover:shadow-xl ${

              isGenerating 

                ? 'bg-slate-400 cursor-not-allowed' 

                : 'bg-indigo-600 hover:bg-indigo-700 hover:-translate-y-1 active:translate-y-0'

            }`}

          >

            {isGenerating ? (

              <Loader2 className="animate-spin" size={24} />

            ) : (

              <Sparkles size={24} />

            )}

            <span>{isGenerating ? 'Generando lista...' : 'Generar Asignaciones para Todos'}</span>

          </button>

        </div>


        {/* Cuadrícula de Resultados */}

        {asignaciones.length > 0 && (

          <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">

            {asignaciones.map((item, index) => (

              <div 

                key={item.id} 

                className="bg-white rounded-2xl p-6 shadow-sm border border-slate-200 hover:shadow-md transition-shadow flex flex-col h-full animate-in fade-in slide-in-from-bottom-4"

                style={{ animationDelay: `${index * 50}ms`, animationFillMode: 'both' }}

              >

                {/* Nombre del Estudiante */}

                <div className="flex items-start space-x-3 mb-4 pb-4 border-b border-slate-100">

                  <div className="bg-blue-100 text-blue-600 p-2 rounded-lg shrink-0">

                    <Users size={20} />

                  </div>

                  <div>

                    <span className="text-xs font-bold text-slate-400 uppercase tracking-wider block mb-1">Estudiante</span>

                    <h3 className="font-bold text-slate-800 leading-tight">

                      {item.nombre}

                    </h3>

                  </div>

                </div>


                <div className="space-y-4 flex-grow">

                  {/* Nivel */}

                  <div className="flex items-start space-x-3">

                    <div className="bg-purple-100 text-purple-600 p-2 rounded-lg shrink-0 mt-1">

                      <GraduationCap size={18} />

                    </div>

                    <div>

                      <span className="text-xs font-bold text-slate-400 uppercase tracking-wider block mb-1">Nivel Asignado</span>

                      <p className="font-medium text-slate-700">{item.nivel}</p>

                    </div>

                  </div>

                </div>

              </div>

            ))}

          </div>

        )}


        {/* Estado inicial vacío */}

        {asignaciones.length === 0 && !isGenerating && (

          <div className="text-center py-20">

            <div className="inline-flex items-center justify-center w-16 h-16 rounded-full bg-slate-200 text-slate-400 mb-4">

              <Users size={32} />

            </div>

            <h3 className="text-xl font-medium text-slate-500">Aún no se han generado las asignaciones</h3>

            <p className="text-slate-400 mt-2">Haz clic en el botón de arriba para comenzar.</p>

          </div>

        )}

      </div>

    </div>

  );

}
